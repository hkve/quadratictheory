        "balk,klji->abij", t^2, tau^202

        "jilk->ijkl", u

        "klji,klba->abij", tau^203, tau^27

    del tau^27

        "baic->iabc", u

    r^2 += einsum(
        "kbac,kjic->abij", tau^204, tau^44
    )

    del tau^44
